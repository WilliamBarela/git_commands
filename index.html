Basic workflow in Git


# check if SSH is working with git and Github:
ssh -T git@github.com

# see the status of your local repository
git status

### basic workflow of using git and github ###

# pull down any changes (git fetch) from Github and merge them with your local (git merge)
git pull

# If you make changes to your local repo, add them to the staging area
git add your_file
# If you need to remove a file from the repo
git rm your_file
# if you need to rename/move your file
git mv your_file to_some_other_filename

# when you are ready to commit what is in the staging area to the local repo
git commit -m "add a meaningful message less than 80 characters"

# if you have more to say, instead of using git commit -m "", use:
# this will open up the system default editor (which can be specified in your ~/.gitconfig 
git commit

# push the changes of your branch back up to Github. 
git push -u origin your_branch

Git log and diff


git log --until=2018-03-05
git log --since=2018-02-27 --until=2018-03-05

# --since is anonymous with --after
git log --since="3 weeks ago" --until="2 days ago"
git log --since=3.weeks --until=2.days

git log --author="Matt"

git log --grep="test"

# show what has commits related to a specific file from a certain time.
git log <SHA_n>.. index.html    # will show all changes since <SHA_n>

# to show what actually changed in each commit:
git log -p

# to show what happened in a specific file for all changes:
git log -p <SHA_n>.. index.html

# to see a list of stats and a summary of all changes
# that is, how many characters have been added and how many have been removed
git log --stat --summary

git log --graph

git log --oneline --graph --all --decorate --stat

# List what files were in a previous commit and the SHA of each file
# These are called tree-ish's
git ls-tree <SHA>/HEAD/HEAD~5

# That version of the file in vim
git show <SHA OF FILE> | vim -

# see the changes between two branches
git diff master..new_feature
git diff --color-words branch1..branch2

# show what has changed between commits (line by line)
git diff <SHA1>..<SHA2>

# check the differences between two branches, the second one one commit before
git diff --color-words branch1..branch2^

# renaming branch
git branch -m new_feature seo_title


Undoing Changes


# Stay on the current branch, just pull out of the repository index.html
# That is, replace the changes I made in my working directory
git checkout -- index.html

# You want to unstage a change without getting rid of your changes.
# Moves what you have in the staging area and puts it in the working directory
git reset HEAD index.html

# you can only change the commit which HEAD is currently pointing to b/c of the SHA
# you can change the files or the message
git commit --ammend -m "Some message"

# how to checkout a previous version of a file
# This moves the file to the staging area
git checkout <some part of the SHA> -- <FILE NAME>

# You can check what the differences are between this one and the current one with
git diff --staged

# If you commit it this reverted commit, include the SHA in the commit message

# If you need to get rid of this old checkout one, you need to do the following:
git reset HEAD resources.html    # moves the changes from staging area to working direcory
git checkout -- resources.html   # gets pull latest commit version of resource.html into WD 

# How to completely revert a commit to a previous version in the repository.
git revert <SHA> 

# moves head pointer, but does not mess with WD or SA
git reset --soft <SHA> 

# moves head pointer, removes everything from the SA, but leaves changes in WD
git reset --mixed <SHA>

# moves head pointer, removes everything from the SA and WD and copies from the old commit
# be sure to ALWAYS SAVE git log because the old object is not immediately garabe collected.
git reset --hard <SHA>

# Remove untracked files (permanently deletes!)
git clean -n    # test run
git clean -f    # removes the files which are in the WORKING DIRECTORY only.

# How to stop tracking changes (.gitignor)
git rm --cached tempfile2.txt                       # removes from tracking
git commit -m "removed temfile from staging index"

# how to track an empty directory 
touch .gitkeep
